const GEMINI_URL = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent';

async function callGemini(system, userMsg, history = []) {
  const key = localStorage.getItem('geminiKey');
  if (!key) throw new Error('Gemini API Key இல்ல! Settings-ல் போடுங்க.');

  const contents = [];
  // Add history
  for (const h of history.slice(-10)) {
    contents.push({ role: h.role === 'assistant' ? 'model' : 'user', parts: [{ text: h.content }] });
  }
  contents.push({ role: 'user', parts: [{ text: userMsg }] });

  const res = await fetch(`${GEMINI_URL}?key=${key}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      system_instruction: { parts: [{ text: system }] },
      contents,
      generationConfig: { maxOutputTokens: 1000 }
    })
  });
  if (!res.ok) {
    const e = await res.json();
    throw new Error(e.error?.message || 'Gemini API Error');
  }
  const data = await res.json();
  return data.candidates[0].content.parts[0].text;
}


async function generateAudio() {
  const key = localStorage.getItem('geminiKey');
  if (!key) { showError('audioError', 'Gemini API Key இல்ல! Settings-ல் போடுங்க.'); return; }
  if (!state.sourceText.trim()) { showError('audioError', 'Content இல்ல! Page 1-ல் text போடுங்க.'); return; }

  showLoading(true);
  document.getElementById('generateBtn').style.display = 'none';
  document.getElementById('audioOutput').classList.remove('visible');

  try {
    const langMap = { tamil: 'Tamil', english: 'English', both: 'Tamil and English mixed' };
    const lenMap = { short: '2-3 minutes key points only', medium: '4-6 minutes balanced', long: '8-10 minutes detailed' };

    let fmtInst = '';
    if (state.selectedVoice === 'solo') {
      fmtInst = 'Return JSON: {"type":"solo","title":"...","paragraphs":["para1","para2"]}';
    } else if (state.selectedVoice === 'duo') {
      fmtInst = 'Return JSON: {"type":"duo","title":"...","lines":[{"speaker":"Arjun","text":"..."},{"speaker":"Priya","text":"..."}]}';
    } else {
      fmtInst = 'Return JSON: {"type":"drama","title":"...","scene":"...","lines":[{"character":"NAME","dialogue":"...","isStage":false}]}';
    }

    const prompt = 'Convert this content into an audio ' + state.selectedVoice + ' script. Language: ' + (langMap[state.selectedLang]||'Tamil') + '. Length: ' + (lenMap[state.selectedLen]||'medium') + '. CONTENT: ' + state.sourceText.slice(0,4000) + '. ' + fmtInst + '. Return ONLY valid JSON.';

    const rawText = await callGemini('You are a script writer. Return only valid JSON.', prompt, []);

    let raw = rawText.trim();
    raw = raw.replace(/^```json/, '').replace(/^```/, '').replace(/```$/, '').trim();
    raw = raw.replace(/,\s*}/g, '}').replace(/,\s*]/g, ']');
    state.generatedScript = JSON.parse(raw);

    showLoading(false);
    document.getElementById('generateBtn').style.display = 'flex';
    document.getElementById('generateBtn').innerHTML = '↺ மீண்டும் உருவாக்கு';
    document.getElementById('audioOutput').classList.add('visible');
    updatePlayUI(false);

  } catch(err) {
    showLoading(false);
    document.getElementById('generateBtn').style.display = 'flex';
    showError('audioError', err.message);
  }
}


async function playAudio() {
  if (!state.generatedScript) return;
  ttsAborted = false;
  updatePlayUI(true);

  const segments = buildSegments(state.generatedScript);
  if (!segments.length) {
    showError('audioError', 'Script இல்ல — மீண்டும் Generate பண்ணுங்க');
    updatePlayUI(false);
    return;
  }

  if (state.selectedVoice === 'solo') {
    const fullText = segments.map(s => s.text).join(' ');
    await playTTSFromBackend(fullText, 'tamil_male');
  } else {
    for (let i = 0; i < segments.length; i++) {
      if (ttsAborted) break;
      const v = segments[i].gender === 'female' ? 'tamil_female' : 'tamil_male';
      await playTTSFromBackend(segments[i].text, v);
      if (!ttsAborted) await new Promise(r => setTimeout(r, 100));
    }
  }

  if (!ttsAborted) updatePlayUI(false);
}

async function playTTSFromBackend(text, voice) {
  if (ttsAborted || !text.trim()) return;
  try {
    const res = await fetch('https://simple-app-uxtf.onrender.com/tts', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ text, voice })
    });
    if (!res.ok) throw new Error('TTS Error');
    const blob = await res.blob();
    const url = URL.createObjectURL(blob);
    return new Promise((resolve) => {
      const audio = new Audio(url);
      audio.onended = () => { URL.revokeObjectURL(url); resolve(); };
      audio.onerror = () => { URL.revokeObjectURL(url); resolve(); };
      currentAudio = audio;
      audio.play().catch(() => resolve());
    });
  } catch(e) {
    if (hasTTS) {
      const u = new SpeechSynthesisUtterance(text);
      u.lang = 'ta-IN';
      return new Promise(r => { u.onend = r; u.onerror = r; window.speechSynthesis.speak(u); });
    }
  }
}


function stopAudio() {
  ttsAborted = true;
  if (currentAudio) { try { currentAudio.pause(); currentAudio = null; } catch(e) {} }
  try { window.speechSynthesis.cancel(); } catch(e) {}
  currentUtterances = [];
  state.isSpeaking = false;
  updatePlayUI(false);
}

function updatePlayUI(playing) {
  const pb = document.getElementById('playBtn');
  const sb = document.getElementById('stopBtn');
  const pi = document.getElementById('playingIndicator');
  if (pb) { pb.style.display = playing ? 'none' : 'flex'; pb.innerHTML = '▶ Play'; }
  if (sb) sb.style.display = playing ? 'flex' : 'none';
  if (pi) pi.style.display = playing ? 'flex' : 'none';
  // Never hide genLoading here — generateAudio controls it
}

// Android Chrome TTS keepalive (prevents cutting off after ~15 sec)
setInterval(() => {
  if (!hasTTS) return;
  if (window.speechSynthesis.speaking && !window.speechSynthesis.paused) {
    try { window.speechSynthesis.pause(); window.speechSynthesis.resume(); } catch(e) {}
  }
}, 12000);

// ===== DOWNLOAD AUDIO =====
async function downloadScript() {
  if (!state.generatedScript) return;
  const dlBtn = document.querySelector('.dl-btn');
  const s = state.generatedScript;

  if (dlBtn) { dlBtn.textContent = '⏳ Recording...'; dlBtn.disabled = true; }

  try {
    // Request mic — user records what TTS speaks through phone speaker
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    const mimeType = (typeof MediaRecorder !== 'undefined' && MediaRecorder.isTypeSupported('audio/webm'))
      ? 'audio/webm' : 'audio/ogg';
    const recorder = new MediaRecorder(stream, { mimeType });
    const chunks = [];
    recorder.ondataavailable = e => { if(e.data.size>0) chunks.push(e.data); };
    recorder.onstop = () => {
      stream.getTracks().forEach(t => t.stop());
      const blob = new Blob(chunks, { type: mimeType });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = 'bookchat-' + s.type + '-' + Date.now() + (mimeType.includes('webm') ? '.webm' : '.ogg');
      document.body.appendChild(a); a.click(); document.body.removeChild(a);
      URL.revokeObjectURL(url);
      if (dlBtn) { dlBtn.textContent = '⬇ Download Audio'; dlBtn.disabled = false; }
    };

    recorder.start(100);
    // Play all segments
    const savedAbort = ttsAborted; ttsAborted = false;
    const segments = buildSegments(state.generatedScript);
    for (let i = 0; i < segments.length; i++) {
      await speakSegment(segments[i], state.selectedLang);
      await new Promise(r => setTimeout(r, 150));
    }
    await new Promise(r => setTimeout(r, 600));
    recorder.stop();
    ttsAborted = savedAbort;

  } catch(err) {
    // Fallback: text script download
    let text = 'BookChat — ' + (s.title||'Script') + '\n' + '='.repeat(40) + '\n\n';
    if (s.type==='solo') text += (s.paragraphs||[]).join('\n\n');
    else if (s.type==='duo') text += (s.lines||[]).map(l=>(l.speaker||'')+': '+(l.text||'')).join('\n\n');
    else if (s.type==='drama') {
      if (s.scene) text += '[' + s.scene + ']\n\n';
      text += (s.lines||[]).map(l => l.isStage ? '(' + l.dialogue + ')' : (l.character||'') + ':\n  ' + (l.dialogue||'')).join('\n\n');
    }
    const blob = new Blob([text],{type:'text/plain'});
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a'); a.href=url;
    a.download='bookchat-script-'+Date.now()+'.txt';
    document.body.appendChild(a); a.click(); document.body.removeChild(a);
    URL.revokeObjectURL(url);
    showError('audioError', 'Mic இல்லாம record ஆகல — Script download ஆச்சு');
    if (dlBtn) { dlBtn.textContent = '⬇ Download Audio'; dlBtn.disabled = false; }
  }
}
